Instalação rápida

Usando CDN — HTML + JS mínimo:

<div id="myTable"></div>

<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/skargrid/dist/skargrid.css">
<script src="https://cdn.jsdelivr.net/npm/skargrid/dist/skargrid.min.js"></script>
<script>
const data = [
    {id:1,nome:'Carlos',cidade:'SP'},
    {id:2,nome:'Fernanda',cidade:'RJ'}
];
const cols = [
    {field:'id',title:'ID'},
    {field:'nome',title:'Nome'},
    {field:'cidade',title:'Cidade'}
];
new Skargrid('myTable', { data, columns: cols, searchable:true, pagination:true, pageSize:5 });
</script>
